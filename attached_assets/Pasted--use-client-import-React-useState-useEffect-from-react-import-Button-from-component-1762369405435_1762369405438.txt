interface CandidateCardProps {
  candidate: any
  onDragStart: (e: React.DragEvent, candidate: any) => void
  onViewCandidate: (candidate: any) => void
  isSelected?: boolean
  onToggleSelect?: (candidateId: number) => void
  bulkActionMode?: boolean
  onActivateLiaPrompt?: (candidateName: string, action: string) => void
  onViewTimeline?: (candidate: any) => void
}

function CandidateCard({ candidate, onDragStart, onViewCandidate, isSelected, onToggleSelect, bulkActionMode, onActivateLiaPrompt, onViewTimeline }: CandidateCardProps) {
  const [showActions, setShowActions] = useState(false)

  // Determinar cor da borda baseado no status
  const getBorderClass = () => {
    if (candidate.approvalPending) {
      return "border-l-4 border-l-yellow-500"
    }
    if (candidate.liaStatus === "triagem_completa") {
      return "border-l-4 border-l-green-500"
    }
    if (candidate.liaStatus === "em_contato") {
      return "border-l-4 border-l-blue-500"
    }
    return "border-l-2 border-l-gray-400"
  }

  // Sistema de avisos contextuais
  // Sistema de avisos contextuais - Tom conversacional da LIA
  const getContextualAlerts = () => {
    const alerts = []

    // Calcular dias sem movimentação (simulado - em produção viria do backend)
    const daysStuck = Math.floor(Math.random() * 15) // 0-14 dias

    // 1. Tempo parado - LIA sugere ação
    if (daysStuck > 7) {
      alerts.push({
        type: 'danger',
        icon: <Clock className="w-3 h-3" />,
        message: `Esse candidato está ${daysStuck} dias parado. Posso movê-lo ou arquivá-lo?`,
        action: 'Ver opções',
        autoAction: `Será arquivado em ${14 - daysStuck} dias se não houver ação`
      })
    } else if (daysStuck > 4) {
      alerts.push({
        type: 'warning',
        icon: <Clock className="w-3 h-3" />,
        message: `Já faz ${daysStuck} dias nesta etapa. Precisa de alguma ajuda?`,
        action: null,
        autoAction: null
      })
    }

    // 2. Feedback pendente - LIA oferece enviar
    if (candidate.status === 'Reprovado' && !candidate.feedbackSent) {
      alerts.push({
        type: 'danger',
        icon: <Mail className="w-3 h-3" />,
        message: 'Posso enviar o feedback de reprovação?',
        action: 'Aprovar envio',
        autoAction: 'Será enviado automaticamente em 3 dias'
      })
    }

    // 3. Dados faltantes - LIA pergunta se deve solicitar
    if (candidate.stage === 'Entrevista' && !candidate.expectedSalary) {
      alerts.push({
        type: 'warning',
        icon: <DollarSign className="w-3 h-3" />,
        message: 'Falta a expectativa salarial. Devo solicitar ao candidato?',
        action: 'Sim, solicitar',
        autoAction: null
      })
    }

    // 4. Disponibilidade - LIA sugere perguntar
    if (candidate.stage === 'Final' && !candidate.availability) {
      alerts.push({
        type: 'warning',
        icon: <Calendar className="w-3 h-3" />,
        message: 'Precisamos da disponibilidade dele. Posso perguntar por WhatsApp?',
        action: 'Enviar mensagem',
        autoAction: null
      })
    }

    // 5. Entrevista próxima - LIA lembra
    if (candidate.status === 'Entrevista marcada' || candidate.status === 'Entrevista agendada') {
      alerts.push({
        type: 'info',
        icon: <Video className="w-3 h-3" />,
        message: `Entrevista ${candidate.lastInteraction}. Já enviei o lembrete!`,
        action: null,
        autoAction: null
      })
    }

    // 6. Risco de abandono - LIA alerta
    if (daysStuck > 10 && candidate.contactStatus === 'não contatado') {
      alerts.push({
        type: 'danger',
        icon: <AlertCircle className="w-3 h-3" />,
        message: 'Ele pode desistir do processo. Posso enviar uma mensagem agora?',
        action: 'Enviar contato',
        autoAction: 'Candidato será marcado como "desistente" em 2 dias'
      })
    }

    // 7. Documentos - LIA oferece solicitar
    if (candidate.stage === 'Aprovados' && Math.random() > 0.5) {
      alerts.push({
        type: 'warning',
        icon: <FileText className="w-3 h-3" />,
        message: 'Faltam documentos para admissão. Devo solicitar a ele?',
        action: 'Sim, solicitar',
        autoAction: 'Solicitação automática em 24h'
      })
    }

    // 8. Triagem pronta - LIA recomenda próxima etapa
    if (candidate.liaStatus === 'triagem_completa' && Math.random() > 0.7) {
      alerts.push({
        type: 'info',
        icon: <CheckCircle className="w-3 h-3" />,
        message: 'Triagem aprovada! Agendar entrevista com ele?',
        action: 'Agendar agora',
        autoAction: null
      })
    }

    return alerts
  }

  const alerts = getContextualAlerts()

  return (
    <Card
      className={`relative cursor-pointer hover:shadow-md transition-all duration-200 ${getBorderClass()} ${candidate.approvalPending ? 'ring-2 ring-yellow-400 ring-opacity-50' : ''} ${isSelected ? 'ring-2 ring-blue-500 bg-blue-50 dark:bg-blue-900/10' : ''}`}
      draggable
      onDragStart={(e) => onDragStart(e, candidate)}
      onClick={(e) => {
        // Se clicar no checkbox, não abre o preview
        if ((e.target as HTMLElement).closest('[data-checkbox]')) {
          return
        }
        onViewCandidate(candidate)
      }}
      onMouseEnter={() => setShowActions(true)}
      onMouseLeave={() => setShowActions(false)}
    >
      <CardContent className="p-3 relative">
        {/* Checkbox de Seleção */}
        {(bulkActionMode || showActions || isSelected) && onToggleSelect && (
          <div
            className="absolute top-2 left-2 z-10"
            data-checkbox
            onClick={(e) => {
              e.stopPropagation()
              onToggleSelect(candidate.id)
            }}
          >
            <div className={`w-5 h-5 rounded border-2 flex items-center justify-center cursor-pointer transition-all ${
              isSelected
                ? 'bg-blue-600 border-blue-600'
                : 'bg-white dark:bg-gray-700 border-gray-300 dark:border-gray-600 hover:border-blue-400'
            }`}>
              {isSelected && (
                <CheckCircle className="w-3.5 h-3.5 text-white" fill="currentColor" />
              )}
            </div>
          </div>
        )}

        {/* Badge de Ação Pendente - Sempre visível */}
        {candidate.approvalPending && (
          <div className={`bg-yellow-100 text-yellow-800 text-xs font-bold px-2 py-1 rounded mb-2 flex items-center justify-between transition-opacity duration-200 ${
            showActions ? 'opacity-0 h-0 overflow-hidden mb-0' : 'opacity-100'
          }`}>
            <span className="flex items-center gap-1">
              <AlertCircle className="w-3.5 h-3.5" />
              AÇÃO NECESSÁRIA
            </span>
            <span className="text-[10px] font-normal">{candidate.pendingSince}</span>
          </div>
        )}

        {/* Header do Card */}
        <div className="flex items-start gap-2 mb-2">
          <Avatar className="w-10 h-10">
            <AvatarImage src={candidate.avatar} alt={candidate.name} />
            <AvatarFallback className="text-xs">{candidate.name.split(' ').map((n: string) => n[0]).join('')}</AvatarFallback>
          </Avatar>

          <div className="flex-1 min-w-0">
            {/* Nome + Ícones */}
            <div className="flex items-center justify-between gap-1 mb-0.5">
              <div className="flex items-center gap-1 flex-1 min-w-0">
                <h4 className="font-semibold text-sm text-gray-900 dark:text-gray-100 truncate">
                  {candidate.name}
                </h4>
                {/* Ícones de ação ao lado do nome */}
                <div className="flex items-center gap-0.5">
                  <Button
                    variant="ghost"
                    size="sm"
                    className="h-5 w-5 p-0 text-blue-600 hover:bg-blue-50 dark:hover:bg-blue-900/20"
                    onClick={(e) => {
                      e.stopPropagation()
                      if (onActivateLiaPrompt) {
                        onActivateLiaPrompt(candidate.name, `Analisar candidato - Score LIA: ${candidate.liaScore}, Match: ${candidate.skillsMatch}%`)
                      }
                    }}
                    title="Ver análise da LIA"
                  >
                    <BrainCircuit className="w-3.5 h-3.5" />
                  </Button>
                  <Button
                    variant="ghost"
                    size="sm"
                    className="h-5 w-5 p-0 text-gray-600 hover:bg-gray-100 dark:hover:bg-gray-700"
                    onClick={(e) => {
                      e.stopPropagation()
                      onViewCandidate(candidate)
                    }}
                    title="Ver perfil do candidato"
                  >
                    <User className="w-3.5 h-3.5" />
                  </Button>
                </div>
              </div>

              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button
                    variant="ghost"
                    size="sm"
                    className={`h-6 w-6 p-0 transition-opacity ${showActions ? 'opacity-100' : 'opacity-0'}`}
                  >
                    <MoreHorizontal className="w-3.5 h-3.5" />
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end" className="w-40">
                  <DropdownMenuItem className="text-xs" onClick={(e) => { e.stopPropagation(); alert(`Ver perfil de ${candidate.name}`) }}>
                    <Eye className="w-3 h-3 mr-2" />
                    Ver perfil
                  </DropdownMenuItem>
                  <DropdownMenuItem className="text-xs">
                    <MessageSquare className="w-3 h-3 mr-2" />
                    Enviar mensagem
                  </DropdownMenuItem>
                  <DropdownMenuItem className="text-xs">
                    <Calendar className="w-3 h-3 mr-2" />
                    Agendar entrevista
                  </DropdownMenuItem>
                  <DropdownMenuSeparator />
                  <DropdownMenuSeparator />
                  <DropdownMenuItem className="text-xs" onClick={(e) => {
                    e.stopPropagation();
                    if (onViewTimeline) {
                      onViewTimeline(candidate)
                    }
                  }}>
                    <Clock className="w-3 h-3 mr-2" />
                    Ver timeline
                  </DropdownMenuItem>
                  <DropdownMenuSeparator />
                  <DropdownMenuItem className="text-xs text-red-600">
                    <AlertCircle className="w-3 h-3 mr-2" />
                    Rejeitar
                  </DropdownMenuItem>
                </DropdownMenuContent>
              </DropdownMenu>
            </div>

            {/* Cargo e Empresa - Logo abaixo do nome */}
            <div className="flex items-center gap-1 text-xs text-gray-600 dark:text-gray-400 mb-1.5">
              <Briefcase className="w-3 h-3 flex-shrink-0" />
              <span className="truncate">{candidate.position || 'UX Designer'} • {candidate.currentCompany || 'TechCorp'}</span>
            </div>

            {/* Localização */}
            <div className="flex items-center gap-1 text-xs text-gray-500 dark:text-gray-400 mb-0.5">
              <MapPin className="w-3 h-3 flex-shrink-0" />
              <span>{candidate.location}</span>
              <span>•</span>
              <span>{candidate.experience}</span>
            </div>

            {/* Última Interação / Status */}
            <div className="flex items-center gap-1 text-xs text-gray-500 dark:text-gray-400">
              <Clock className="w-3 h-3 flex-shrink-0" />
              <span className="truncate">{candidate.lastInteraction}</span>
            </div>
          </div>
        </div>

        {/* Separador */}
        <div className="h-px bg-gray-200 dark:bg-gray-700 my-2"></div>

        {/* Scores - Compactos */}
        <div className="flex items-center gap-3 mb-2">
          <div className="flex items-center gap-1">
            <BrainCircuit className="w-3.5 h-3.5 text-blue-600" />
            <span className="text-xs font-bold text-blue-900 dark:text-blue-100">{candidate.liaScore}</span>
          </div>
          <div className="flex items-center gap-1">
            <Target className="w-3.5 h-3.5 text-green-600" />
            <span className="text-xs font-bold text-green-900 dark:text-green-100">{candidate.skillsMatch}%</span>
          </div>
          {candidate.npsScore && (
            <div className="flex items-center gap-1">
              <ThumbsUp className="w-3.5 h-3.5 text-purple-600" />
              <span className={`text-xs font-bold ${
                candidate.npsScore >= 80 ? 'text-green-700 dark:text-green-300' :
                candidate.npsScore >= 60 ? 'text-yellow-700 dark:text-yellow-300' :
                'text-red-700 dark:text-red-300'
              }`}>
                {candidate.npsScore}
              </span>
            </div>
          )}
        </div>

        {/* Badges de Status - Agrupadas */}
        <div className="flex flex-wrap gap-1 mb-2">
          {/* Status da LIA */}
          {candidate.liaStatus && (
            <Badge
              variant="outline"
              className={`text-[10px] px-2 py-0.5 ${
                candidate.liaStatus === 'aguardando_aprovacao_contato' ? 'bg-yellow-50 text-yellow-700 border-yellow-300' :
                candidate.liaStatus === 'em_contato' ? 'bg-blue-50 text-blue-700 border-blue-300' :
                candidate.liaStatus === 'triagem_completa' ? 'bg-green-50 text-green-700 border-green-300' :
                candidate.liaStatus === 'aguardando_decisao' ? 'bg-orange-50 text-orange-700 border-orange-300' :
                'bg-gray-50 text-gray-600 border-gray-200'
              }`}
            >
              {candidate.liaStatus === 'aguardando_aprovacao_contato' && <Zap className="w-2.5 h-2.5 mr-1" />}
              {candidate.liaStatus === 'em_contato' && <Phone className="w-2.5 h-2.5 mr-1" />}
              {candidate.liaStatus === 'triagem_completa' && <CheckCircle className="w-2.5 h-2.5 mr-1" />}
              {candidate.liaStatus === 'aguardando_decisao' && <Target className="w-2.5 h-2.5 mr-1" />}
              {candidate.liaStatus === 'aguardando_aprovacao_contato' && 'Aprovar Contato'}
              {candidate.liaStatus === 'em_contato' && 'LIA em Contato'}

          {/* Status de Contato */}
          {candidate.contactStatus && (

                candidate.contactStatus === 'feedback enviado' ? 'bg-green-50 text-green-700 border-green-200' :

                'bg-gray-50 text-gray-600 border-gray-200'
              }`}
            >

              {candidate.contactStatus === 'tentando contato' && <Clock className="w-2.5 h-2.5 mr-1" />}
              {candidate.contactStatus === 'não contatado' && <User className="w-2.5 h-2.5 mr-1" />}

            </div>
            <div className="flex-1 h-px bg-gray-300 dark:bg-gray-600"></div>
          </div>
        )}

        {/* Dados da Triagem - Aparecem no Hover */}

            <div className="text-[10px] font-medium text-green-700 dark:text-green-400 mb-1">✓ Triagem Completa</div>
            <div className="grid grid-cols-2 gap-x-2 gap-y-1 text-[9px] text-green-600 dark:text-green-500">
              <div>• Mobilidade: {candidate.triageData.mobility}</div>
              <div>• Salário: {candidate.triageData.salary}</div>
              <div>• Disponib.: {candidate.triageData.availability}</div>
              <div>• Interesse: {candidate.triageData.interest}</div>
            </div>
          </div>
        )}

        {/* Sugestão da LIA - Aparece no Hover */}
        {candidate.liaSuggestion && (
          <div className={`bg-blue-50 dark:bg-blue-900/20 rounded p-2 mb-2 transition-all duration-300 ${
            showActions ? 'opacity-100 max-h-40' : 'opacity-0 max-h-0 p-0 mb-0 overflow-hidden'
          }`}>
            <div className="text-[10px] text-blue-700 dark:text-blue-400">
              <BrainCircuit className="w-3 h-3 inline mr-1" />
              <strong>Parecer LIA:</strong> {candidate.liaSuggestion}
            </div>
          </div>
        )}

        {/* Avisos Contextuais */}
        {alerts.length > 0 && (
          <div className="space-y-1 mb-2">
            {alerts.map((contextAlert, index) => (
              <div
                key={index}
                className={`rounded p-2 flex items-start gap-2 text-[10px] ${
                  contextAlert.type === 'danger'
                    ? 'bg-red-50 dark:bg-red-900/20 text-red-700 dark:text-red-400 border border-red-200 dark:border-red-800'
                  : contextAlert.type === 'warning'
                    ? 'bg-orange-50 dark:bg-orange-900/20 text-orange-700 dark:text-orange-400 border border-orange-200 dark:border-orange-800'
                  : contextAlert.type === 'info'
                    ? 'bg-blue-50 dark:bg-blue-900/20 text-blue-700 dark:text-blue-400 border border-blue-200 dark:border-blue-800'
                    : 'bg-gray-50 dark:bg-gray-900/20 text-gray-700 dark:text-gray-400 border border-gray-200 dark:border-gray-800'
                }`}
              >
                <div className="mt-0.5">{contextAlert.icon}</div>
                <div className="flex-1">
                  <div className="font-medium">{contextAlert.message}</div>
                  {contextAlert.autoAction && (
                    <div className="mt-1 text-[9px] opacity-70 italic">
                      ⏱️ {contextAlert.autoAction}
                    </div>
                  )}
                  {contextAlert.action && (
                    <button
                      onClick={(e) => {
                        e.stopPropagation()
                        if (onActivateLiaPrompt) {
                          onActivateLiaPrompt(candidate.name, contextAlert.action)
                        }
                      }}
                      className={`mt-1 text-[9px] font-semibold underline hover:no-underline ${
                        contextAlert.type === 'danger' ? 'text-red-800 dark:text-red-300' :
                        contextAlert.type === 'warning' ? 'text-orange-800 dark:text-orange-300' :
                        'text-blue-800 dark:text-blue-300'
                      }`}
                    >
                      {contextAlert.action} →
                    </button>
                  )}
                </div>
              </div>
            ))}
          </div>
        )}

        {/* Ações - Aparecem SOMENTE no hover */}
        {(candidate.approvalPending || (candidate.status === 'Reprovado' && !candidate.feedbackSent)) && showActions && (
          <div className="flex gap-1 flex-wrap">
            {candidate.approvalPending && (
              <>
                <Button
                  size="sm"
                  variant="outline"
                  className="h-6 px-2 text-[10px] bg-green-50 text-green-700 border-green-300 hover:bg-green-100"
                  onClick={(e) => {
                    e.stopPropagation()
                    if (onActivateLiaPrompt) {
                      const action = candidate.liaStatus === 'aguardando_aprovacao_contato'
                        ? 'Aprovar contato com candidato'
                        : 'Aprovar próximo passo para candidato'
                      onActivateLiaPrompt(candidate.name, action)
                    }
                  }}
                >
                  <CheckCircle className="w-3 h-3 mr-1" />
                  Aprovar
                </Button>
                <Button
                  size="sm"
                  variant="outline"
                  className="h-6 px-2 text-[10px] text-red-600 border-red-300 hover:bg-red-50"
                  onClick={(e) => {
                    e.stopPropagation()
                    if (onActivateLiaPrompt) {
                      onActivateLiaPrompt(candidate.name, 'Rejeitar candidato e enviar feedback')
                    }
                  }}
                >
                  <X className="w-3 h-3 mr-1" />
                  Rejeitar
                </Button>
              </>
            )}

            {candidate.status === 'Reprovado' && !candidate.feedbackSent && (
              <Button
                size="sm"
                variant="outline"
                className="h-6 px-2 text-[10px] text-orange-600 border-orange-300"
                onClick={(e) => {
                  e.stopPropagation()
                  if (onActivateLiaPrompt) {
                    onActivateLiaPrompt(candidate.name, 'Enviar feedback de reprovação para candidato')
                  }
                }}
              >
                <Mail className="w-3 h-3 mr-1" />
                Feedback
              </Button>
            )}
          </div>
        )}
      </CardContent>
    </Card>
  )
}

interface CandidateTableRowProps {
  candidate: any
  onCandidateClick: (candidate: any) => void
  isSelected?: boolean
  onToggleSelect?: (candidateId: number) => void
  calculateRanking: (candidate: any) => number
  getAlerts: (candidate: any) => any[]
  getUrgency: (ranking: number) => any
  onViewBigFive?: (candidate: any) => void
}

function CandidateTableRow({ candidate, onCandidateClick, isSelected, onToggleSelect, calculateRanking, getAlerts, getUrgency, onViewBigFive }: CandidateTableRowProps) {
  const [showActions, setShowActions] = useState(false)
  const ranking = calculateRanking(candidate)
  const alerts = getAlerts(candidate)
  const urgency = getUrgency(ranking)

  return (
    <tr
      className={`border-b border-gray-200 dark:border-gray-700 hover:bg-gray-50 dark:hover:bg-gray-800 cursor-pointer transition-colors ${
        isSelected ? 'bg-blue-50 dark:bg-blue-900/10' : ''
      }`}
      onMouseEnter={() => setShowActions(true)}
      onMouseLeave={() => setShowActions(false)}
      onClick={() => onCandidateClick(candidate)}
    >
      {/* Checkbox */}
      <td className="px-4 py-3" onClick={(e) => e.stopPropagation()}>
        <div
          onClick={() => onToggleSelect && onToggleSelect(candidate.id)}
          className="cursor-pointer"
        >
          <div className={`w-5 h-5 rounded border-2 flex items-center justify-center transition-all ${
            isSelected
              ? 'bg-blue-600 border-blue-600'
              : 'bg-white dark:bg-gray-700 border-gray-300 dark:border-gray-600 hover:border-blue-400'
          }`}>
            {isSelected && (
              <CheckCircle className="w-3.5 h-3.5 text-white" fill="currentColor" />
            )}
          </div>
        </div>
      </td>

      {/* ID Column */}
      <td className="px-4 py-3">
        <div className="text-xs font-mono text-gray-600 dark:text-gray-400">
          {candidate.id}
        </div>
      </td>

      {/* Nota LIA Geral */}
      <td className="px-3 py-3">
        <div className="flex flex-col items-center gap-1">
          <div className="flex items-center gap-1">
            <BrainCircuit className={`w-3 h-3 ${
              urgency.level === 'excellent' ? 'text-green-600' :
              urgency.level === 'great' ? 'text-blue-600' :
              urgency.level === 'good' ? 'text-cyan-600' :
              urgency.level === 'average' ? 'text-yellow-600' :
              urgency.level === 'below' ? 'text-orange-600' :
              'text-red-600'
            }`} />
            <span className={`text-base font-semibold ${
              urgency.level === 'excellent' ? 'text-green-600' :
              urgency.level === 'great' ? 'text-blue-600' :
              urgency.level === 'good' ? 'text-cyan-600' :
              urgency.level === 'average' ? 'text-yellow-600' :
              urgency.level === 'below' ? 'text-orange-600' :
              'text-red-600'
            }`}>
              {ranking}
            </span>
          </div>

            {urgency.label}

              <DropdownMenuSeparator />
              <DropdownMenuItem className="text-xs text-red-600">
                <AlertCircle className="w-3 h-3 mr-2" />
                Rejeitar
              </DropdownMenuItem>
            </DropdownMenuContent>
          </DropdownMenu>
        </div>
      </td>
    </tr>
  )
}

  // Calculate Nota LIA Geral (0-100)
  const calculateNotaLiaGeral = (candidate: any) => {
    // COMPONENTES DA NOTA (cada um 0-100)

    // 1. Score LIA CV (análise curricular vs requisitos da vaga) - peso 25%
    const scoreLiaCV = candidate.skillsMatch || 0 // já vem 0-100

    // 2. Score LIA Triagem (pré-requisitos técnicos + comportamentais) - peso 30%
    const scoreLiaTriagem = (candidate.liaScore || 0) * 10 // converte 0-10 para 0-100

    // 3. Score Teste Técnico (quando aplicável) - peso 25%
    const scoreTesteTecnico = candidate.technicalTestScore || 0 // 0-100

    // 4. Score Teste Inglês (quando aplicável) - peso 20%
    const scoreTesteIngles = candidate.englishTestScore || 0 // 0-100

    // CÁLCULO DA NOTA BASE (média ponderada)
    const notaBase = (
      (scoreLiaCV * 0.25) +
      (scoreLiaTriagem * 0.30) +
      (scoreTesteTecnico * 0.25) +
      (scoreTesteIngles * 0.20)
    )

    // BÔNUS DE URGÊNCIA (+10 pontos)
    let bonusUrgencia = 0
    if (candidate.approvalPending && candidate.liaStatus === 'aguardando_aprovacao_contato') {
      bonusUrgencia = 10 // Crítico
    } else if (candidate.approvalPending && candidate.liaStatus === 'triagem_completa') {
      bonusUrgencia = 8 // Alto
    } else if (candidate.approvalPending) {
      bonusUrgencia = 5 // Médio
    }

    // BÔNUS POR ETAPA AVANÇADA (+5 pontos)
    let bonusEtapa = 0
    if (candidate.stage === 'Final') {
      bonusEtapa = 5
    } else if (candidate.stage === 'Entrevista') {
      bonusEtapa = 3
    }

    // BÔNUS POR TRIAGEM COMPLETA E APROVADA (+5 pontos)
    let bonusTriagem = 0
    if (candidate.triageComplete && candidate.triageData) {
      const isApproved = candidate.triageData.mobility === 'OK' &&
                        candidate.triageData.salary !== 'Acima do budget' &&
                        candidate.triageData.interest !== 'Baixo'
      bonusTriagem = isApproved ? 5 : 2
    } else if (candidate.liaStatus === 'triagem_completa') {
      bonusTriagem = 5
    }

    // PENALIZAÇÃO POR TEMPO PARADO (-0.5 pontos por dia, máx -10)
    const daysStuck = Math.floor(Math.random() * 15) // Em produção, viria do backend
    const penalizacaoTempo = Math.min(daysStuck * 0.5, 10)

    // NOTA FINAL (0-100)
    let notaFinal = notaBase + bonusUrgencia + bonusEtapa + bonusTriagem - penalizacaoTempo

    // Garantir que fica entre 0 e 100
    notaFinal = Math.max(0, Math.min(100, notaFinal))

    return Math.round(notaFinal * 10) / 10 // arredonda para 1 casa decimal
  }

  // Get urgency level for visual display based on Nota LIA Geral

    if (nota >= 80) return { level: 'great', label: 'Muito Bom', emoji: '⭐', color: 'from-blue-500 to-blue-600' }
    if (nota >= 70) return { level: 'good', label: 'Bom', emoji: '👍', color: 'from-cyan-500 to-blue-500' }
    if (nota >= 60) return { level: 'average', label: 'Regular', emoji: '📋', color: 'from-yellow-400 to-orange-500' }
    if (nota >= 50) return { level: 'below', label: 'Abaixo', emoji: '⚠️', color: 'from-orange-500 to-red-500' }
    return { level: 'low', label: 'Baixo', emoji: '❌', color: 'from-red-500 to-red-600' }

        }

          setIsListening(false)

    }
  }

          break
        case 'notaLiaGeral':

          bVal = calculateNotaLiaGeral(b)
          break
        default:
          aVal = calculateNotaLiaGeral(a)
          bVal = calculateNotaLiaGeral(b)
      }

      if (aVal < bVal) return tableSortDirection === 'asc' ? -1 : 1
      if (aVal > bVal) return tableSortDirection === 'asc' ? 1 : -1
      return 0

        duration: '3 dias',
        action: 'Triagem completa - Aprovado',
        actor: 'LIA',
        icon: <CheckCircle className="w-4 h-4" />,
        color: 'bg-green-500'
      },
      candidate.status === 'Entrevista marcada' && {
        stage: 'Agendamento',
        date: new Date(baseDate.getTime() + 7 * 24 * 60 * 60 * 1000).toISOString(),
        duration: '2 dias',
        action: 'Entrevista agendada',
        actor: candidate.recruiter || 'Recrutador',
        icon: <Calendar className="w-4 h-4" />,
        color: 'bg-yellow-500'
      },
      candidate.status === 'Reprovado' && {
        stage: 'Reprovado',
        date: new Date(baseDate.getTime() + 10 * 24 * 60 * 60 * 1000).toISOString(),
        duration: '3 dias',
        action: `Reprovado - ${candidate.rejectionReason || 'Motivo não informado'}`,
        actor: currentJob.manager,
        icon: <X className="w-4 h-4" />,
        color: 'bg-red-500'
      }
    ].filter(Boolean)
  }

  const handleViewTimeline = (candidate: any) => {
    setTimelineCandidate(candidate)
    setShowTimeline(true)
  }

  // Gerar sugestões de busca inteligentes
  const getSearchSuggestions = () => {
    if (!searchTerm.trim()) {
      return {
        history: searchHistory.slice(0, 3),
        candidates: [],
        skills: [],
        companies: [],
        locations: [],
        filters: []
      }
    }

    const term = searchTerm.toLowerCase()
    const allCandidates = Object.values(candidatesData).flat()

    // Candidatos que correspondem
    const matchingCandidates = allCandidates
      .filter((c: any) => c.name.toLowerCase().includes(term))
      .slice(0, 3)
      .map((c: any) => ({
        type: 'candidate' as const,
        label: c.name,
        subtitle: `${c.position || 'UX Designer'} • Score: ${c.liaScore}`,
        avatar: c.avatar
      }))

    // Skills que correspondem
    const allSkills = new Set<string>()
    allCandidates.forEach((c: any) => {
      c.skills?.forEach((s: string) => allSkills.add(s))
    })
    const matchingSkills = Array.from(allSkills)
      .filter(s => s.toLowerCase().includes(term))
      .slice(0, 3)
      .map(s => ({
        type: 'skill' as const,
        label: s,
        subtitle: `${allCandidates.filter((c: any) => c.skills?.includes(s)).length} candidatos`
      }))

    // Empresas que correspondem
    const allCompanies = new Set<string>()
    allCandidates.forEach((c: any) => {
      if (c.currentCompany) allCompanies.add(c.currentCompany)
    })
    const matchingCompanies = Array.from(allCompanies)
      .filter(co => co.toLowerCase().includes(term))
      .slice(0, 2)
      .map(co => ({
        type: 'company' as const,
        label: co,
        subtitle: `${allCandidates.filter((c: any) => c.currentCompany === co).length} candidatos`
      }))

    // Localizações que correspondem
    const allLocations = new Set<string>()
    allCandidates.forEach((c: any) => {
      if (c.location) allLocations.add(c.location)
    })
    const matchingLocations = Array.from(allLocations)
      .filter(loc => loc.toLowerCase().includes(term))
      .slice(0, 2)
      .map(loc => ({
        type: 'location' as const,
        label: loc,
        subtitle: `${allCandidates.filter((c: any) => c.location === loc).length} candidatos`
      }))

    // Filtros inteligentes
    const smartFilters = [
      { query: 'score lia > 8', label: 'Score LIA > 8', count: allCandidates.filter((c: any) => c.liaScore > 8).length },
      { query: 'match > 90', label: 'Match > 90%', count: allCandidates.filter((c: any) => c.skillsMatch > 90).length },
      { query: 'aprovação pendente', label: 'Aprovação Pendente', count: allCandidates.filter((c: any) => c.approvalPending).length },
      { query: 'triagem completa', label: 'Triagem Completa', count: allCandidates.filter((c: any) => c.triageComplete).length },
    ].filter(f => f.label.toLowerCase().includes(term))
      .map(f => ({
        type: 'filter' as const,
        label: f.label,
        subtitle: `${f.count} candidatos`,
        query: f.query
      }))

    return {
      history: [],
      candidates: matchingCandidates,
      skills: matchingSkills,
      companies: matchingCompanies,
      locations: matchingLocations,
      filters: smartFilters
    }
  }

  const searchSuggestions = getSearchSuggestions()
  const allSuggestions = [
    ...searchSuggestions.history.map(h => ({ type: 'history' as const, label: h, subtitle: 'Busca recente' })),
    ...searchSuggestions.candidates,
    ...searchSuggestions.skills,
    ...searchSuggestions.companies,
    ...searchSuggestions.locations,
    ...searchSuggestions.filters
  ]

  const handleSearchKeyDown = (e: React.KeyboardEvent) => {
    if (!showSearchSuggestions) return

    if (e.key === 'ArrowDown') {
      e.preventDefault()
      setSelectedSuggestionIndex(prev =>
        prev < allSuggestions.length - 1 ? prev + 1 : prev
      )
    } else if (e.key === 'ArrowUp') {
      e.preventDefault()
      setSelectedSuggestionIndex(prev => prev > 0 ? prev - 1 : -1)
    } else if (e.key === 'Enter' && selectedSuggestionIndex >= 0) {
      e.preventDefault()
      const suggestion = allSuggestions[selectedSuggestionIndex]
      if (suggestion) {
        const searchValue = (suggestion as any).query || suggestion.label
        setSearchTerm(searchValue)
        if (!searchHistory.includes(searchValue)) {
          setSearchHistory(prev => [searchValue, ...prev.slice(0, 4)])
        }
        setShowSearchSuggestions(false)
        setSelectedSuggestionIndex(-1)
      }
    } else if (e.key === 'Escape') {
      setShowSearchSuggestions(false)
      setSelectedSuggestionIndex(-1)
    }
  }

  const handleSuggestionClick = (suggestion: any) => {
    const searchValue = suggestion.query || suggestion.label
    setSearchTerm(searchValue)
    if (!searchHistory.includes(searchValue)) {
      setSearchHistory(prev => [searchValue, ...prev.slice(0, 4)])
    }
    setShowSearchSuggestions(false)
    setSelectedSuggestionIndex(-1)
  }

  // Highlight matching text
  const highlightMatch = (text: string, query: string) => {
    if (!query) return text
    const index = text.toLowerCase().indexOf(query.toLowerCase())
    if (index === -1) return text

    return (
      <>
        {text.slice(0, index)}
        <span className="font-bold text-blue-600 dark:text-blue-400">
          {text.slice(index, index + query.length)}
        </span>
        {text.slice(index + query.length)}
      </>
    )
  }

  // Função para ativar o prompt da LIA com ação específica
  const activateLiaPrompt = (candidateName: string, action: string) => {
    const promptText = `${action} - ${candidateName}`
    setLiaPromptValue(promptText)

    setLiaHighlight(true)

    setTimeout(() => setLiaHighlight(false), 1000)

    // Scroll suave para o prompt e foca no input
    setTimeout(() => {
      liaInputRef.current?.scrollIntoView({ behavior: 'smooth', block: 'center' })
      liaInputRef.current?.focus()
    }, 100)
  }

    e.dataTransfer.setData('candidateId', candidate.id)
  }

  const handleViewCandidate = (candidate: any) => {
    setSelectedCandidate(candidate)
    setShowCandidateModal(true)
  }

        newSet.delete(candidateId)

        newSet.add(candidateId)

      if (newSet.size > 0) {
        setShowExpandedLIA(true)
        setLiaHighlight(true)
        setTimeout(() => setLiaHighlight(false), 1000)
      }

      return newSet
    })
  }

    Object.values(candidatesData).forEach(candidates => {
      candidates.forEach((c: any) => allIds.add(c.id))
    })
    setSelectedCandidatesForBatch(allIds)
    setBulkActionMode(true)
    setShowExpandedLIA(true)
    setLiaHighlight(true)
    setTimeout(() => setLiaHighlight(false), 1000)
  }

    setBulkActionMode(false)
  }

  return (
    <div className="h-screen bg-gray-50 dark:bg-gray-950 flex flex-col overflow-hidden">
      {/* Header Principal */}
      <div className="bg-gray-50 dark:bg-gray-950 border-b border-gray-200 dark:border-gray-700">
        <div className="p-4">
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-3">
              <Button variant="ghost" size="sm" onClick={onBack} className="gap-2">
                <ArrowLeft className="w-4 h-4" />
                Voltar
              </Button>

              <div>
                <h1 className="text-lg font-semibold text-gray-900 dark:text-gray-100">
                  {currentJob.title}

                <p className="text-sm text-gray-500">
                  {currentJob.jobId} • {currentJob.department} • {currentJob.location}
                </p>

              {/* Métricas Compactas */}

                <div className="flex items-center gap-1.5">

                <div className="flex items-center gap-1.5">
                  <span className="text-xs text-gray-500">Triagem:</span>
                  <span className="text-sm font-bold text-blue-600">{currentJob.funnel.screening}</span>
                </div>
                <div className="w-px h-4 bg-gray-300 dark:bg-gray-600"></div>
                <div className="flex items-center gap-1.5">
                  <span className="text-xs text-gray-500">Entrevista:</span>
                  <span className="text-sm font-bold text-yellow-600">{currentJob.funnel.interview}</span>
                </div>
                <div className="w-px h-4 bg-gray-300 dark:bg-gray-600"></div>
                <div className="flex items-center gap-1.5">
                  <span className="text-xs text-gray-500">Final:</span>
                  <span className="text-sm font-bold text-purple-600">{currentJob.funnel.final}</span>
                </div>
                <div className="w-px h-4 bg-gray-300 dark:bg-gray-600"></div>
                <div className="flex items-center gap-1.5">
                  <span className="text-xs text-gray-500">NPS:</span>
                  <span className="text-sm font-bold text-indigo-600">{currentJob.nps}</span>
                </div>
              </div>

              {/* Botões de Ação */}
              <Button

                size="sm"
                className="gap-2"
                onClick={() => setShowMetricsDashboard(!showMetricsDashboard)}
              >
                <TrendingUp className="w-4 h-4" />
                Métricas LIA

              <Button

                className="gap-2"

              >
                <FileText className="w-4 h-4" />
                Relatório

                <Share2 className="w-4 h-4" />

                    Preferências

            </div>
          </div>

          {/* Layout Otimizado: Prompt da LIA + Controles */}
          <div className="grid grid-cols-2 gap-4 mb-4">
            {/* Prompt da LIA - Metade Esquerda */}
            <div className={`bg-blue-50 dark:bg-blue-900/20 rounded-lg py-2 px-3 border transition-all duration-300 ${
              liaHighlight
                ? 'border-blue-400 dark:border-blue-600 shadow-lg shadow-blue-200/50 dark:shadow-blue-800/50 scale-[1.01]'
                : 'border-blue-200 dark:border-blue-800'
            }`}>
            <div className="flex items-center gap-2">
              <div className="w-7 h-7 bg-blue-600 dark:bg-blue-700 rounded-lg flex items-center justify-center flex-shrink-0">
                <BrainCircuit className="w-3.5 h-3.5 text-white" />

                <input
                  ref={liaInputRef}
                  type="text"
                  placeholder="Peça à LIA para analisar candidatos, identificar gargalos, mover candidatos entre etapas, agendar entrevistas..."
                  className="w-full bg-transparent border-none outline-none text-sm text-gray-700 dark:text-gray-300 placeholder-gray-500 dark:placeholder-gray-400"
                  value={liaPromptValue}
                  onChange={(e) => setLiaPromptValue(e.target.value)}
                  onFocus={() => setShowExpandedLIA(true)}
                  onKeyDown={(e) => {
                    if (e.key === 'Enter' && liaPromptValue.trim()) {
                      // Executar comando da LIA
                      alert(`Executando: ${liaPromptValue}`)
                    }
                  }}
                />
              </div>
              <div className="flex items-center gap-2 flex-shrink-0">
                <Badge variant="outline" className="text-[11px]">
                  ● {currentJob.funnel.total} candidatos

                {/* Candidatos Selecionados */}
                {selectedCandidatesForBatch.size > 0 && (
                  <div className="mb-2 p-2 bg-blue-100 dark:bg-blue-900/30 rounded-lg">
                    <div className="flex items-center justify-between mb-1">
                      <p className="text-xs font-medium text-blue-700 dark:text-blue-300">
                        {selectedCandidatesForBatch.size} candidato(s) selecionado(s)
                      </p>
                      <Button
                        size="sm"
                        variant="ghost"
                        className="h-5 text-[10px] px-2 text-blue-600"

                        <X className="w-3 h-3 mr-1" />
                        Limpar

                  </p>
                  <div className="flex flex-wrap gap-1">

                      <>
                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue(`Mover ${selectedCandidatesForBatch.size} candidatos para próxima etapa`)}
                        >
                          <ArrowUpDown className="w-3 h-3 mr-1" />
                          Mover Etapa
                        </Button>
                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue(`Agendar entrevistas para ${selectedCandidatesForBatch.size} candidatos`)}
                        >
                          <Calendar className="w-3 h-3 mr-1" />
                          Agendar
                        </Button>
                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue(`Enviar mensagem para ${selectedCandidatesForBatch.size} candidatos`)}
                        >
                          <MessageCircle className="w-3 h-3 mr-1" />
                          Mensagem

                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue(`Comparar perfis dos ${selectedCandidatesForBatch.size} candidatos selecionados`)}
                        >

                          Comparar

                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"

                        >

                        >
                          <AlertCircle className="w-3 h-3 mr-1" />
                          Riscos

                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue("Sugerir próximas ações para candidatos parados")}
                        >
                          <Zap className="w-3 h-3 mr-1" />
                          Próximas Ações

                        <Button
                          size="sm"
                          variant="outline"
                          className="h-6 text-[10px] px-2"
                          onClick={() => setLiaPromptValue("Top 5 candidatos por score LIA")}
                        >
                          <Star className="w-3 h-3 mr-1" />
                          Top Candidatos

                          className="h-6 text-[10px] px-2"

                          onClick={() => setLiaPromptValue(`Analisar compatibilidade dos candidatos selecionados com a vaga ${currentJob.title}`)}

                          onClick={() => setLiaPromptValue(`Lembrar ${currentJob.recruiter} sobre candidatos aguardando aprovação`)}

                    <button
                      className="p-2 bg-blue-50 dark:bg-blue-900/20 rounded text-xs hover:bg-blue-100 dark:hover:bg-blue-900/30 transition-colors cursor-pointer"
                      onClick={() => setLiaPromptValue("Qualidade do pipeline atual vs histórico")}
                    >
                      <div className="text-blue-700 dark:text-blue-300 font-medium text-[10px]">Qualidade</div>
                      <div className="text-blue-600 dark:text-blue-400 text-[9px]">Pipeline</div>
                    </button>
                    <button
                      className="p-2 bg-orange-50 dark:bg-orange-900/20 rounded text-xs hover:bg-orange-100 dark:hover:bg-orange-900/30 transition-colors cursor-pointer"
                      onClick={() => setLiaPromptValue("Otimizar distribuição de candidatos entre etapas")}
                    >
                      <div className="text-orange-700 dark:text-orange-300 font-medium text-[10px]">Otimizar</div>
                      <div className="text-orange-600 dark:text-orange-400 text-[9px]">Distribuição</div>
                    </button>
                    <button
                      className="p-2 bg-indigo-50 dark:bg-indigo-900/20 rounded text-xs hover:bg-indigo-100 dark:hover:bg-indigo-900/30 transition-colors cursor-pointer"
                      onClick={() => setLiaPromptValue("Comparar com benchmarks de mercado")}
                    >
                      <div className="text-indigo-700 dark:text-indigo-300 font-medium text-[10px]">Benchmark</div>
                      <div className="text-indigo-600 dark:text-indigo-400 text-[9px]">Mercado</div>
                    </button>
                  </div>
                </div>

                {/* Métricas em Tempo Real */}
                <div className="mt-3 pt-3 border-t border-gray-200 dark:border-gray-700">
                  <p className="text-xs text-gray-600 dark:text-gray-400 mb-2">
                    📊 Métricas do funil:
                  </p>
                  <div className="grid grid-cols-2 gap-2">
                    <div className="p-2 bg-green-50 dark:bg-green-900/20 rounded text-xs">
                      <span className="text-green-700 dark:text-green-300 font-medium">Taxa de Conversão</span>
                      <p className="text-green-600 dark:text-green-400 text-[10px] mt-1">
                        Triagem → Entrevista: {currentJob.funnel.screening > 0 ? Math.round((currentJob.funnel.interview / currentJob.funnel.screening) * 100) : 0}%
                      </p>
                    </div>
                    <div className="p-2 bg-blue-50 dark:bg-blue-900/20 rounded text-xs">
                      <span className="text-blue-700 dark:text-blue-300 font-medium">Tempo Médio</span>
                      <p className="text-blue-600 dark:text-blue-400 text-[10px] mt-1">
                        Por etapa: {currentJob.avgTimePerStage?.screening || 3} dias
                      </p>
                    </div>
                    <div className="p-2 bg-purple-50 dark:bg-purple-900/20 rounded text-xs">
                      <span className="text-purple-700 dark:text-purple-300 font-medium">Aprovações</span>
                      <p className="text-purple-600 dark:text-purple-400 text-[10px] mt-1">
                        {Object.values(candidatesData).flat().filter((c: any) => c.approvalPending).length} pendentes
                      </p>
                    </div>
                    <div className="p-2 bg-orange-50 dark:bg-orange-900/20 rounded text-xs">
                      <span className="text-orange-700 dark:text-orange-300 font-medium">Urgência</span>
                      <p className="text-orange-600 dark:text-orange-400 text-[10px] mt-1">
                        Nível {currentJob.urgencyLevel}/5
                      </p>
                    </div>
                  </div>
                </div>
              </div>
            )}
            </div>

            {/* Controles - Metade Direita */}
            <div className="space-y-3">
              {/* Busca e Filtros */}
              <div className="flex items-center gap-2">
                <div className="relative flex-1">
                  <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-400 z-10" />
                  <input
                    ref={searchInputRef}
                    type="text"
                    placeholder="Buscar candidatos, skills, empresas..."
                    value={searchTerm}
                    onChange={(e) => {
                      setSearchTerm(e.target.value)
                      setShowSearchSuggestions(true)
                      setSelectedSuggestionIndex(-1)
                    }}
                    onFocus={() => setShowSearchSuggestions(true)}
                    onBlur={() => {
                      // Delay to allow click on suggestion
                      setTimeout(() => setShowSearchSuggestions(false), 200)
                    }}
                    onKeyDown={handleSearchKeyDown}
                    className="w-full pl-10 pr-10 py-2 text-sm border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent bg-white dark:bg-gray-800"
                  />
                  <div className="absolute right-3 top-1/2 -translate-y-1/2 flex items-center gap-1">
                    {/* Busca por Voz */}
                    {voiceSearchSupported && (
                      <button
                        onClick={toggleVoiceSearch}
                        className={`p-1 rounded-full transition-colors ${
                          isListening
                            ? 'bg-red-500 text-white animate-pulse'
                            : 'text-gray-400 hover:text-blue-600 hover:bg-blue-50 dark:hover:bg-blue-900/20'
                        }`}
                        title={isListening ? 'Gravando... Clique para parar' : 'Buscar por voz'}
                      >
                        {isListening ? (
                          <div className="relative">
                            <Bell className="w-4 h-4" />
                            <span className="absolute -top-1 -right-1 w-2 h-2 bg-white rounded-full"></span>
                          </div>
                        ) : (
                          <Bell className="w-4 h-4" />
                        )}
                      </button>
                    )}

                    {/* Limpar busca */}
                    {searchTerm && (
                      <button
                        onClick={() => {
                          setSearchTerm('')
                          setShowSearchSuggestions(false)
                        }}
                        className="text-gray-400 hover:text-gray-600 dark:hover:text-gray-300"
                      >
                        <X className="w-4 h-4" />
                      </button>
                    )}
                  </div>

                  {/* Autocomplete Suggestions */}
                  {showSearchSuggestions && allSuggestions.length > 0 && (
                    <div className="absolute top-full left-0 right-0 mt-1 bg-white dark:bg-gray-800 border border-gray-200 dark:border-gray-700 rounded-lg shadow-lg max-h-96 overflow-y-auto z-50">
                      {/* Histórico de buscas */}
                      {searchSuggestions.history.length > 0 && (
                        <div className="border-b border-gray-200 dark:border-gray-700">
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <Clock className="w-3 h-3" />
                            Buscas recentes
                          </div>
                          {searchSuggestions.history.map((item, index) => (
                            <button
                              key={`history-${index}`}
                              onClick={() => handleSuggestionClick({ label: item, type: 'history' })}
                              className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                selectedSuggestionIndex === index ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                              }`}
                            >
                              <Clock className="w-3 h-3 text-gray-400" />
                              <span className="flex-1">{item}</span>
                            </button>
                          ))}
                        </div>
                      )}

                      {/* Candidatos */}
                      {searchSuggestions.candidates.length > 0 && (
                        <div className="border-b border-gray-200 dark:border-gray-700">
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <User className="w-3 h-3" />
                            Candidatos
                          </div>
                          {searchSuggestions.candidates.map((item, index) => {
                            const globalIndex = searchSuggestions.history.length + index
                            return (
                              <button
                                key={`candidate-${index}`}
                                onClick={() => handleSuggestionClick(item)}
                                className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                  selectedSuggestionIndex === globalIndex ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                                }`}
                              >
                                <Avatar className="w-6 h-6">
                                  <AvatarImage src={item.avatar} />
                                  <AvatarFallback className="text-xs">
                                    {item.label.split(' ').map((n: string) => n[0]).join('')}
                                  </AvatarFallback>
                                </Avatar>
                                <div className="flex-1 min-w-0">
                                  <div className="text-gray-900 dark:text-gray-100 truncate">
                                    {highlightMatch(item.label, searchTerm)}
                                  </div>
                                  <div className="text-xs text-gray-500 dark:text-gray-400 truncate">
                                    {item.subtitle}
                                  </div>
                                </div>
                              </button>
                            )
                          })}
                        </div>
                      )}

                      {/* Skills */}
                      {searchSuggestions.skills.length > 0 && (
                        <div className="border-b border-gray-200 dark:border-gray-700">
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <Target className="w-3 h-3" />
                            Skills
                          </div>
                          {searchSuggestions.skills.map((item, index) => {
                            const globalIndex = searchSuggestions.history.length + searchSuggestions.candidates.length + index
                            return (
                              <button
                                key={`skill-${index}`}
                                onClick={() => handleSuggestionClick(item)}
                                className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                  selectedSuggestionIndex === globalIndex ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                                }`}
                              >
                                <Badge variant="outline" className="text-xs">
                                  {highlightMatch(item.label, searchTerm)}
                                </Badge>
                                <span className="text-xs text-gray-500 dark:text-gray-400">{item.subtitle}</span>
                              </button>
                            )
                          })}
                        </div>
                      )}

                      {/* Empresas */}
                      {searchSuggestions.companies.length > 0 && (
                        <div className="border-b border-gray-200 dark:border-gray-700">
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <Building className="w-3 h-3" />
                            Empresas
                          </div>
                          {searchSuggestions.companies.map((item, index) => {
                            const globalIndex = searchSuggestions.history.length + searchSuggestions.candidates.length + searchSuggestions.skills.length + index
                            return (
                              <button
                                key={`company-${index}`}
                                onClick={() => handleSuggestionClick(item)}
                                className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                  selectedSuggestionIndex === globalIndex ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                                }`}
                              >
                                <Building className="w-4 h-4 text-gray-400" />
                                <div className="flex-1">
                                  <div className="text-gray-900 dark:text-gray-100">
                                    {highlightMatch(item.label, searchTerm)}
                                  </div>
                                  <div className="text-xs text-gray-500 dark:text-gray-400">{item.subtitle}</div>
                                </div>
                              </button>
                            )
                          })}
                        </div>
                      )}

                      {/* Localizações */}
                      {searchSuggestions.locations.length > 0 && (
                        <div className="border-b border-gray-200 dark:border-gray-700">
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <MapPin className="w-3 h-3" />
                            Localizações
                          </div>
                          {searchSuggestions.locations.map((item, index) => {
                            const globalIndex = searchSuggestions.history.length + searchSuggestions.candidates.length + searchSuggestions.skills.length + searchSuggestions.companies.length + index
                            return (
                              <button
                                key={`location-${index}`}
                                onClick={() => handleSuggestionClick(item)}
                                className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                  selectedSuggestionIndex === globalIndex ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                                }`}
                              >
                                <MapPin className="w-4 h-4 text-gray-400" />
                                <div className="flex-1">
                                  <div className="text-gray-900 dark:text-gray-100">
                                    {highlightMatch(item.label, searchTerm)}
                                  </div>
                                  <div className="text-xs text-gray-500 dark:text-gray-400">{item.subtitle}</div>
                                </div>
                              </button>
                            )
                          })}
                        </div>
                      )}

                      {/* Filtros Inteligentes */}
                      {searchSuggestions.filters.length > 0 && (
                        <div>
                          <div className="px-3 py-2 text-xs font-medium text-gray-500 dark:text-gray-400 flex items-center gap-1">
                            <Filter className="w-3 h-3" />
                            Filtros sugeridos
                          </div>
                          {searchSuggestions.filters.map((item, index) => {
                            const globalIndex = searchSuggestions.history.length + searchSuggestions.candidates.length + searchSuggestions.skills.length + searchSuggestions.companies.length + searchSuggestions.locations.length + index
                            return (
                              <button
                                key={`filter-${index}`}
                                onClick={() => handleSuggestionClick(item)}
                                className={`w-full px-3 py-2 text-left text-sm hover:bg-gray-50 dark:hover:bg-gray-700 flex items-center gap-2 ${
                                  selectedSuggestionIndex === globalIndex ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                                }`}
                              >
                                <Zap className="w-4 h-4 text-blue-500" />
                                <div className="flex-1">
                                  <div className="text-gray-900 dark:text-gray-100">
                                    {highlightMatch(item.label, searchTerm)}
                                  </div>
                                  <div className="text-xs text-gray-500 dark:text-gray-400">{item.subtitle}</div>
                                </div>
                              </button>
                            )
                          })}
                        </div>
                      )}

                      {/* Dica de navegação */}
                      <div className="px-3 py-2 text-xs text-gray-400 dark:text-gray-500 border-t border-gray-200 dark:border-gray-700 flex items-center gap-2">
                        <kbd className="px-1.5 py-0.5 text-xs bg-gray-100 dark:bg-gray-700 rounded">↑↓</kbd>
                        <span>navegar</span>
                        <kbd className="px-1.5 py-0.5 text-xs bg-gray-100 dark:bg-gray-700 rounded">Enter</kbd>
                        <span>selecionar</span>
                        <kbd className="px-1.5 py-0.5 text-xs bg-gray-100 dark:bg-gray-700 rounded">Esc</kbd>
                        <span>fechar</span>
                      </div>
                    </div>
                  )}
                </div>
                <Button variant="outline" size="sm" className="gap-2 whitespace-nowrap">
                  <Filter className="w-4 h-4" />
                  Filtros
                </Button>
              </div>

              {/* Ordenação e Modos de Visualização */}
              <div className="flex items-center justify-between gap-2">
                <DropdownMenu>
                  <DropdownMenuTrigger asChild>
                    <Button variant="outline" size="sm" className="gap-2">
                      <ArrowUpDown className="w-4 h-4" />
                      Ordenar
                    </Button>
                  </DropdownMenuTrigger>
                  <DropdownMenuContent>
                    {sortingOptions.map(option => (
                      <DropdownMenuItem
                        key={option.id}
                        onClick={() => setSelectedSort(option.id)}
                      >
                        {option.icon}
                        <span className="ml-2">{option.label}</span>
                      </DropdownMenuItem>
                    ))}
                  </DropdownMenuContent>
                </DropdownMenu>

                {/* Modos de Visualização */}
                <div className="flex items-center gap-2">
                  <Button
                    variant={viewMode === 'kanban' ? 'default' : 'outline'}
                    size="sm"
                    onClick={() => setViewMode('kanban')}
                    className="gap-2"
                  >
                    <Columns className="w-4 h-4" />
                    Kanban
                  </Button>
                  <Button
                    variant={viewMode === 'table' ? 'default' : 'outline'}
                    size="sm"
                    onClick={() => setViewMode('table')}
                    className="gap-2"
                  >
                    <Table className="w-4 h-4" />
                    Tabela
                  </Button>
                </div>
              </div>

              {/* Ações em Lote - Se houver seleção */}
              {selectedCandidatesForBatch.size > 0 && (
                <div className="flex items-center gap-2 p-2 bg-blue-50 dark:bg-blue-900/20 rounded-lg border border-blue-200 dark:border-blue-800">
                  <Badge variant="secondary" className="gap-1">
                    {selectedCandidatesForBatch.size} selecionado{selectedCandidatesForBatch.size > 1 ? 's' : ''}
                  </Badge>
                  <Button variant="outline" size="sm" onClick={deselectAllCandidates} className="h-7">
                    <X className="w-3 h-3" />
                  </Button>
                  <Button variant="default" size="sm" className="gap-2 h-7 flex-1">
                    <Zap className="w-3 h-3" />
                    Ações em Lote
                  </Button>
                </div>
              )}
            </div>
          </div>
        </div>
      </div>

      {/* Dashboard de Métricas da LIA */}
      {showMetricsDashboard && (
        <div className="bg-white dark:bg-gray-900 border-b border-gray-200 dark:border-gray-700 p-4">
          <LiaMetricsDashboard candidates={getAllCandidates()} />
        </div>
      )}

      {/* Kanban Columns */}
      {viewMode === 'kanban' && (
        <div className="flex-1 overflow-x-auto overflow-y-hidden bg-gray-50 dark:bg-gray-950 p-4">
          <div className="flex gap-4 h-full min-w-max">
            {pipelineStages.filter(s => s.visible).map((stage) => (
              <div key={stage.id} className="flex-shrink-0 w-80">
                <div className="bg-white dark:bg-gray-800 rounded-lg shadow-sm h-full flex flex-col">
                  {/* Column Header */}
                  <div className="p-3 border-b border-gray-200 dark:border-gray-700">
                    <div className="flex items-center justify-between mb-2">
                      <div className="flex items-center gap-2">
                        <h3 className="font-semibold text-sm">{stage.name}</h3>
                        <Badge variant="secondary" className="text-xs">
                          {candidatesData[stage.id as keyof typeof candidatesData]?.length || 0}
                        </Badge>
                      </div>
                      <DropdownMenu>
                        <DropdownMenuTrigger asChild>
                          <Button variant="ghost" size="sm" className="h-6 w-6 p-0">
                            <MoreHorizontal className="w-4 h-4" />
                          </Button>
                        </DropdownMenuTrigger>
                        <DropdownMenuContent align="end">
                          <DropdownMenuItem>
                            <Plus className="w-4 h-4 mr-2" />
                            Adicionar candidato
                          </DropdownMenuItem>
                          <DropdownMenuItem>
                            <CheckCircle className="w-4 h-4 mr-2" />
                            Selecionar todos
                          </DropdownMenuItem>
                          <DropdownMenuSeparator />
                          <DropdownMenuItem>
                            <Settings className="w-4 h-4 mr-2" />
                            Configurar coluna
                          </DropdownMenuItem>
                        </DropdownMenuContent>
                      </DropdownMenu>
                    </div>
                  </div>

                  {/* Column Content */}
                  <div className="flex-1 overflow-y-auto p-3 space-y-3">
                    {(candidatesData[stage.id as keyof typeof candidatesData] || []).map((candidate: any) => (
                      <CandidateCard
                        key={candidate.id}
                        candidate={candidate}
                        onDragStart={handleDragStart}
                        onViewCandidate={handleViewCandidate}
                        isSelected={selectedCandidatesForBatch.has(candidate.id)}
                        onToggleSelect={toggleCandidateSelection}
                        bulkActionMode={bulkActionMode}
                        onActivateLiaPrompt={activateLiaPrompt}
                        onViewTimeline={handleViewTimeline}
                      />
                    ))}
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>
      )}

      {/* Table View */}
      {viewMode === 'table' && (
        <div className="flex-1 overflow-auto bg-gray-50 dark:bg-gray-950 flex flex-col">
          <div className="p-4 space-y-4">
            {/* Filtros de Etapa - Pipeline Flow Design */}
            <div className="bg-white dark:bg-gray-900 rounded-lg border border-gray-200 dark:border-gray-700 p-4">
              <div className="flex items-center justify-between mb-3">
                <div className="flex items-center gap-3">
                  <div className="w-8 h-8 bg-blue-50 dark:bg-blue-900/20 rounded-lg flex items-center justify-center">
                    <Columns className="w-4 h-4 text-blue-600 dark:text-blue-400" />
                  </div>
                  <div>
                    <h4 className="text-sm font-semibold text-gray-900 dark:text-gray-100">
                      Filtrar por Etapa
                    </h4>
                    <p className="text-xs text-gray-500 dark:text-gray-400">
                      {tableStageFilter.length > 0
                        ? `${tableStageFilter.length} etapa${tableStageFilter.length > 1 ? 's' : ''} selecionada${tableStageFilter.length > 1 ? 's' : ''}`
                        : 'Todas as etapas'
                      }
                    </p>
                  </div>
                </div>
                {tableStageFilter.length > 0 && (
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={clearStageFilters}
                    className="h-7 text-xs gap-1"
                  >
                    <X className="w-3 h-3" />
                    Limpar
                  </Button>
                )}
              </div>

              {/* Pipeline Flow */}
              <div className="flex items-center gap-2">
                {pipelineStages.map((stage, index) => {
                  const isSelected = tableStageFilter.includes(stage.name)
                  const count = getStageCount(stage.name)

                  return (
                    <React.Fragment key={stage.id}>
                      <button
                        onClick={() => toggleStageFilter(stage.name)}
                        className={`
                          flex-1 relative
                          transition-all duration-200
                          ${isSelected ? 'scale-[1.02]' : 'hover:scale-[1.01]'}
                        `}
                      >
                        <div className={`
                          rounded-lg px-4 py-3
                          border
                          transition-all duration-200
                          ${isSelected
                            ? `${stage.color} border-blue-400 dark:border-blue-500 shadow-sm`
                            : 'bg-white dark:bg-gray-800 border-gray-200 dark:border-gray-700 hover:border-gray-300 dark:hover:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-750'
                          }
                        `}>
                          {/* Stage Name */}
                          <div className={`
                            text-xs font-semibold mb-1
                            ${isSelected ? '' : 'text-gray-600 dark:text-gray-400'}
                          `}>
                            {stage.name}
                          </div>

                          {/* Count */}
                          <div className="flex items-baseline gap-1">
                            <span className={`
                              text-2xl font-bold
                              ${isSelected
                                ? 'text-current'
                                : 'text-gray-900 dark:text-gray-100'
                              }
                            `}>
                              {count}
                            </span>
                            <span className={`
                              text-[10px] font-medium
                              ${isSelected
                                ? 'text-current opacity-70'
                                : 'text-gray-500 dark:text-gray-400'
                              }
                            `}>
                              {count === 1 ? 'candidato' : 'candidatos'}
                            </span>
                          </div>

                          {/* Progress Bar */}
                          <div className="mt-2 h-1 bg-gray-200 dark:bg-gray-700 rounded-full overflow-hidden">
                            <div
                              className={`h-full transition-all duration-500 ${
                                isSelected
                                  ? 'bg-current'
                                  : 'bg-gray-400 dark:bg-gray-600'
                              }`}
                              style={{
                                width: `${Math.min(100, (count / getAllCandidates().length) * 100 * 3)}%`
                              }}
                            />
                          </div>
                        </div>

                        {/* Selection Indicator */}
                        {isSelected && (
                          <div className="absolute -top-1.5 -right-1.5">
                            <div className="w-5 h-5 bg-blue-600 dark:bg-blue-500 rounded-full flex items-center justify-center shadow-md">
                              <CheckCircle className="w-3 h-3 text-white" fill="currentColor" />
                            </div>
                          </div>
                        )}
                      </button>

                      {/* Connector */}
                      {index < pipelineStages.length - 1 && (
                        <div className="flex-shrink-0 text-gray-300 dark:text-gray-600">
                          <svg width="16" height="16" viewBox="0 0 16 16" fill="none">
                            <path d="M6 4L10 8L6 12" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/>
                          </svg>
                        </div>
                      )}
                    </React.Fragment>
                  )
                })}
              </div>

              {/* Quick Stats */}
              <div className="mt-3 pt-3 border-t border-gray-200 dark:border-gray-700 flex items-center justify-between text-xs">
                <div className="flex items-center gap-4">
                  <div className="flex items-center gap-1.5">
                    <div className="w-2 h-2 bg-green-500 rounded-full"></div>
                    <span className="text-gray-600 dark:text-gray-400">
                      Taxa de Conversão:
                    </span>
                    <span className="font-semibold text-gray-900 dark:text-gray-100">
                      {Math.round((pipelineStages.find(s => s.name === 'Final')?.count || 0) / (getAllCandidates().length || 1) * 100)}%
                    </span>
                  </div>
                  <div className="flex items-center gap-1.5">
                    <div className="w-2 h-2 bg-blue-500 rounded-full"></div>
                    <span className="text-gray-600 dark:text-gray-400">
                      Em Processo:
                    </span>
                    <span className="font-semibold text-gray-900 dark:text-gray-100">
                      {getAllCandidates().filter((c: any) =>
                        ['Triagem', 'Entrevista', 'Final'].includes(c.stage)
                      ).length}
                    </span>
                  </div>
                </div>
                <div className="text-gray-500 dark:text-gray-400">
                  Clique nas etapas para filtrar
                </div>
              </div>
            </div>

            {/* Tabela */}
            <div className="bg-white dark:bg-gray-900 border border-gray-200 dark:border-gray-700 rounded-lg overflow-hidden">
              <table className="w-full">
                <thead className="bg-white dark:bg-gray-900 border-b-2 border-gray-200 dark:border-gray-700">
                  <tr>
                    {/* Checkbox Header */}
                    <th className="px-4 py-3 text-left">
                      <div
                        onClick={() => {
                          const paginatedData = getPaginatedCandidates()
                          const allIds = new Set(paginatedData.candidates.map((c: any) => c.id))
                          const allSelected = paginatedData.candidates.every((c: any) => selectedCandidatesForBatch.has(c.id))

                          if (allSelected) {
                            // Deselect all from current page
                            setSelectedCandidatesForBatch(prev => {
                              const newSet = new Set(prev)
                              paginatedData.candidates.forEach((c: any) => newSet.delete(c.id))
                              return newSet
                            })
                          } else {
                            // Select all from current page
                            setSelectedCandidatesForBatch(prev => {
                              const newSet = new Set(prev)
                              paginatedData.candidates.forEach((c: any) => newSet.add(c.id))
                              return newSet
                            })
                          }
                          setBulkActionMode(selectedCandidatesForBatch.size > 0)
                        }}
                        className="cursor-pointer"
                      >
                        <div className={`w-5 h-5 rounded border-2 flex items-center justify-center transition-all ${
                          getPaginatedCandidates().candidates.length > 0 && getPaginatedCandidates().candidates.every((c: any) => selectedCandidatesForBatch.has(c.id))
                            ? 'bg-blue-600 border-blue-600'
                            : 'bg-white dark:bg-gray-700 border-gray-300 dark:border-gray-600 hover:border-blue-400'
                        }`}>
                          {getPaginatedCandidates().candidates.length > 0 && getPaginatedCandidates().candidates.every((c: any) => selectedCandidatesForBatch.has(c.id)) && (
                            <CheckCircle className="w-3.5 h-3.5 text-white" fill="currentColor" />
                          )}
                        </div>
                      </div>
                    </th>

                    {/* ID Column */}
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      ID
                    </th>

                    {/* Nota LIA Geral */}
                    <th
                      className="px-3 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('notaLiaGeral')}
                    >
                      <div className="flex items-center gap-2">
                        <BrainCircuit className="w-3.5 h-3.5 text-blue-600" />
                        Nota LIA
                        {tableSortColumn === 'notaLiaGeral' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    {/* Score LIA Triagem */}
                    <th
                      className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('scoreLiaTriagem')}
                    >
                      <div className="flex items-center gap-2">
                        Triagem LIA
                        {tableSortColumn === 'scoreLiaTriagem' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    {/* Score LIA CV */}
                    <th
                      className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('scoreLiaCV')}
                    >
                      <div className="flex items-center gap-2">
                        Score LIA CV
                        {tableSortColumn === 'scoreLiaCV' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    {/* Teste Técnico */}
                    <th
                      className="px-3 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('testeTecnico')}
                    >
                      <div className="flex items-center gap-2">
                        <FileText className="w-3.5 h-3.5 text-purple-600" />
                        Técnico
                        {tableSortColumn === 'testeTecnico' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    {/* Teste Inglês */}
                    <th
                      className="px-3 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('testeIngles')}
                    >
                      <div className="flex items-center gap-2">
                        <Globe className="w-3.5 h-3.5 text-indigo-600" />
                        Inglês
                        {tableSortColumn === 'testeIngles' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    {/* Assessment Big Five */}
                    <th className="px-3 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      <div className="flex items-center gap-2">
                        <TrendingUp className="w-3.5 h-3.5 text-purple-600" />
                        Big Five
                      </div>
                    </th>

                    {/* Alertas LIA */}
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      <div className="flex items-center gap-2">
                        <Bell className="w-3.5 h-3.5 text-orange-600" />
                        Alertas LIA
                      </div>
                    </th>

                    {/* Candidato */}
                    <th
                      className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('name')}
                    >
                      <div className="flex items-center gap-2">
                        Candidato
                        {tableSortColumn === 'name' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>

                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Cargo
                    </th>
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Empresa
                    </th>
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Salário Atual
                    </th>
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Expectativa
                    </th>
                    <th
                      className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('location')}
                    >
                      <div className="flex items-center gap-2">
                        Localização
                        {tableSortColumn === 'location' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>
                    <th
                      className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300 cursor-pointer hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors"
                      onClick={() => handleTableSort('stage')}
                    >
                      <div className="flex items-center gap-2">
                        Etapa
                        {tableSortColumn === 'stage' && (
                          tableSortDirection === 'asc' ? <SortAsc className="w-3 h-3" /> : <SortDesc className="w-3 h-3" />
                        )}
                      </div>
                    </th>
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Status
                    </th>
                    <th className="px-4 py-3 text-left text-xs font-semibold text-gray-700 dark:text-gray-300">
                      Ações
                    </th>
                  </tr>
                </thead>
                <tbody className="bg-white dark:bg-gray-900 divide-y divide-gray-200 dark:divide-gray-700">
                  {getPaginatedCandidates().candidates.map((candidate: any) => (
                    <CandidateTableRow
                      key={candidate.id}
                      candidate={candidate}
                      onCandidateClick={handleViewCandidate}
                      isSelected={selectedCandidatesForBatch.has(candidate.id)}
                      onToggleSelect={toggleCandidateSelection}
                      calculateRanking={calculateNotaLiaGeral}
                      getAlerts={getLiaAlerts}
                      getUrgency={getUrgencyLevel}
                      onViewBigFive={(cand) => {
                        setBigFiveCandidate(cand)
                        setShowBigFive(true)
                      }}
                    />
                  ))}
                </tbody>
              </table>
            </div>

            {/* Paginação */}
            {getPaginatedCandidates().totalPages > 1 && (
              <div className="bg-white dark:bg-gray-900 border border-gray-200 dark:border-gray-700 rounded-lg p-3">
                <div className="flex items-center justify-between">
                  <div className="text-sm text-gray-600 dark:text-gray-400">
                    Mostrando {((currentPage - 1) * itemsPerPage) + 1} - {Math.min(currentPage * itemsPerPage, getPaginatedCandidates().total)} de {getPaginatedCandidates().total} candidatos
                  </div>
                  <div className="flex items-center gap-2">
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setCurrentPage(1)}
                      disabled={currentPage === 1}
                      className="h-8"
                    >
                      Primeira
                    </Button>
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setCurrentPage(prev => Math.max(1, prev - 1))}
                      disabled={currentPage === 1}
                      className="h-8"
                    >
                      Anterior
                    </Button>

                    {/* Page numbers */}
                    <div className="flex items-center gap-1">
                      {Array.from({ length: getPaginatedCandidates().totalPages }, (_, i) => i + 1)
                        .filter(page => {
                          // Show first page, last page, current page, and pages around current
                          return page === 1 ||
                                 page === getPaginatedCandidates().totalPages ||
                                 (page >= currentPage - 1 && page <= currentPage + 1)
                        })
                        .map((page, index, array) => (
                          <React.Fragment key={page}>
                            {/* Show ellipsis if there's a gap */}
                            {index > 0 && page - array[index - 1] > 1 && (
                              <span className="px-2 text-gray-400">...</span>
                            )}
                            <Button
                              variant={currentPage === page ? 'default' : 'outline'}
                              size="sm"
                              onClick={() => setCurrentPage(page)}
                              className="h-8 w-8 p-0"
                            >
                              {page}
                            </Button>
                          </React.Fragment>
                        ))
                      }
                    </div>

                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setCurrentPage(prev => Math.min(getPaginatedCandidates().totalPages, prev + 1))}
                      disabled={currentPage === getPaginatedCandidates().totalPages}
                      className="h-8"
                    >
                      Próxima
                    </Button>
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setCurrentPage(getPaginatedCandidates().totalPages)}
                      disabled={currentPage === getPaginatedCandidates().totalPages}
                      className="h-8"
                    >
                      Última
                    </Button>
                  </div>
                </div>
              </div>
            )}
          </div>
        </div>
      )}

      {/* Report Modal */}
      {showReport && (
        <JobReportModal
          job={currentJob}
          isOpen={showReport}
          onClose={() => setShowReport(false)}
        />
      )}

      {/* Big Five Modal */}
      <BigFiveModal
        isOpen={showBigFive}
        onClose={() => {
          setShowBigFive(false)
          setBigFiveCandidate(null)
        }}
        candidate={bigFiveCandidate}
      />

      {/* Timeline Modal */}
      {showTimeline && timelineCandidate && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 backdrop-blur-sm">
          <div className="bg-white dark:bg-gray-900 rounded-lg shadow-xl max-w-2xl w-full mx-4 max-h-[90vh] overflow-hidden flex flex-col">
            {/* Header */}
            <div className="p-4 border-b border-gray-200 dark:border-gray-700 flex items-center justify-between">
              <div className="flex items-center gap-3">
                <Avatar className="w-10 h-10">
                  <AvatarImage src={timelineCandidate.avatar} />
                  <AvatarFallback>
                    {timelineCandidate.name.split(' ').map((n: string) => n[0]).join('')}
                  </AvatarFallback>
                </Avatar>
                <div>
                  <h3 className="text-lg font-semibold text-gray-900 dark:text-gray-100">
                    {timelineCandidate.name}
                  </h3>
                  <p className="text-sm text-gray-500">
                    Timeline de Movimentações
                  </p>
                </div>
              </div>
              <Button
                variant="ghost"
                size="sm"
                onClick={() => {
                  setShowTimeline(false)
                  setTimelineCandidate(null)
                }}
              >
                <X className="w-4 h-4" />
              </Button>
            </div>

            {/* Timeline Content */}
            <div className="flex-1 overflow-y-auto p-6">
              {/* Resumo Rápido */}
              <div className="grid grid-cols-3 gap-4 mb-6">
                <div className="p-3 bg-blue-50 dark:bg-blue-900/20 rounded-lg">
                  <div className="text-xs text-blue-600 dark:text-blue-400 mb-1">Tempo Total</div>
                  <div className="text-xl font-bold text-blue-900 dark:text-blue-100">
                    {Math.floor((new Date().getTime() - new Date(timelineCandidate.appliedDate || '2025-03-01').getTime()) / (1000 * 60 * 60 * 24))} dias
                  </div>
                </div>
                <div className="p-3 bg-green-50 dark:bg-green-900/20 rounded-lg">
                  <div className="text-xs text-green-600 dark:text-green-400 mb-1">Etapas</div>
                  <div className="text-xl font-bold text-green-900 dark:text-green-100">
                    {getCandidateTimeline(timelineCandidate).length}
                  </div>
                </div>
                <div className="p-3 bg-purple-50 dark:bg-purple-900/20 rounded-lg">
                  <div className="text-xs text-purple-600 dark:text-purple-400 mb-1">Score LIA</div>
                  <div className="text-xl font-bold text-purple-900 dark:text-purple-100">
                    {timelineCandidate.liaScore}
                  </div>
                </div>
              </div>

              {/* Timeline */}
              <div className="relative">
                {/* Linha vertical */}
                <div className="absolute left-6 top-0 bottom-0 w-0.5 bg-gray-200 dark:bg-gray-700"></div>

                {/* Eventos */}
                <div className="space-y-6">
                  {getCandidateTimeline(timelineCandidate).map((event: any, index: number) => (
                    <div key={index} className="relative pl-16">
                      {/* Ícone */}
                      <div className={`absolute left-0 w-12 h-12 rounded-full ${event.color} flex items-center justify-center text-white shadow-lg`}>
                        {event.icon}
                      </div>

                      {/* Conteúdo */}
                      <div className="bg-gray-50 dark:bg-gray-800 rounded-lg p-4 shadow-sm border border-gray-200 dark:border-gray-700">
                        <div className="flex items-start justify-between mb-2">
                          <div>
                            <h4 className="font-semibold text-sm text-gray-900 dark:text-gray-100">
                              {event.stage}
                            </h4>
                            <p className="text-xs text-gray-500 dark:text-gray-400">
                              {new Date(event.date).toLocaleDateString('pt-BR', {
                                day: '2-digit',
                                month: 'long',
                                year: 'numeric',
                                hour: '2-digit',
                                minute: '2-digit'
                              })}
                            </p>
                          </div>
                          <Badge variant="outline" className="text-xs">
                            {event.duration}
                          </Badge>
                        </div>
                        <p className="text-sm text-gray-700 dark:text-gray-300 mb-2">
                          {event.action}
                        </p>
                        <div className="flex items-center gap-2 text-xs text-gray-500 dark:text-gray-400">
                          <User className="w-3 h-3" />
                          <span>{event.actor}</span>
                        </div>
                      </div>
                    </div>
                  ))}

                  {/* Status Atual */}
                  <div className="relative pl-16">
                    <div className="absolute left-0 w-12 h-12 rounded-full bg-gradient-to-br from-blue-500 to-purple-500 flex items-center justify-center text-white shadow-lg animate-pulse">
                      <Target className="w-4 h-4" />
                    </div>
                    <div className="bg-gradient-to-br from-blue-50 to-purple-50 dark:from-blue-900/20 dark:to-purple-900/20 rounded-lg p-4 border-2 border-blue-200 dark:border-blue-800">
                      <h4 className="font-semibold text-sm text-gray-900 dark:text-gray-100 mb-1">
                        Status Atual
                      </h4>
                      <p className="text-sm text-gray-700 dark:text-gray-300">
                        {timelineCandidate.status} • {timelineCandidate.lastInteraction}
                      </p>
                      {timelineCandidate.liaSuggestion && (
                        <div className="mt-2 p-2 bg-white/50 dark:bg-gray-900/50 rounded text-xs">
                          <strong className="text-blue-600 dark:text-blue-400">Parecer LIA:</strong>
                          <p className="text-gray-600 dark:text-gray-400 mt-1">{timelineCandidate.liaSuggestion}</p>
                        </div>
                      )}
                    </div>
                  </div>
                </div>
              </div>

              {/* Próximas Ações Sugeridas */}
              <div className="mt-6 p-4 bg-yellow-50 dark:bg-yellow-900/20 rounded-lg border border-yellow-200 dark:border-yellow-800">
                <h4 className="font-semibold text-sm text-yellow-900 dark:text-yellow-100 mb-2 flex items-center gap-2">
                  <Zap className="w-4 h-4" />
                  Próximas Ações Sugeridas
                </h4>
                <ul className="space-y-2">
                  {timelineCandidate.approvalPending && (
                    <li className="text-sm text-yellow-800 dark:text-yellow-200 flex items-start gap-2">
                      <CheckCircle className="w-4 h-4 mt-0.5 flex-shrink-0" />
                      <span>Aprovar próximo passo no processo seletivo</span>
                    </li>
                  )}
                  {timelineCandidate.triageComplete && !timelineCandidate.status?.includes('Entrevista') && (
                    <li className="text-sm text-yellow-800 dark:text-yellow-200 flex items-start gap-2">
                      <Calendar className="w-4 h-4 mt-0.5 flex-shrink-0" />
                      <span>Agendar entrevista técnica</span>
                    </li>
                  )}
                  <li className="text-sm text-yellow-800 dark:text-yellow-200 flex items-start gap-2">
                    <MessageCircle className="w-4 h-4 mt-0.5 flex-shrink-0" />
                    <span>Enviar atualização sobre o processo</span>
                  </li>
                </ul>
              </div>
            </div>

            {/* Footer com Ações */}
            <div className="p-4 border-t border-gray-200 dark:border-gray-700 flex items-center justify-between">
              <Button
                variant="outline"
                size="sm"
                onClick={() => {
                  activateLiaPrompt(timelineCandidate.name, 'Analisar timeline e sugerir próximas ações')
                  setShowTimeline(false)
                }}
                className="gap-2"
              >
                <BrainCircuit className="w-4 h-4" />
                Analisar com LIA
              </Button>
              <div className="flex gap-2">
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => handleViewCandidate(timelineCandidate)}
                >
                  Ver Perfil Completo
                </Button>
                <Button
                  variant="default"
                  size="sm"
                  onClick={() => {
                    setShowTimeline(false)
                    setTimelineCandidate(null)
                  }}
                >
                  Fechar
                </Button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}
