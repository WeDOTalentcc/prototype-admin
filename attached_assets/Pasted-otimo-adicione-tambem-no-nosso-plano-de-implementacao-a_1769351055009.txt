class CompensationPolicy(Base):
    """
    Política de compensação por nível/cargo cadastrada pela empresa.
    Permite análise proativa de alinhamento durante criação de vaga.
    """
    __tablename__ = "compensation_policies"
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    company_id = Column(UUID(as_uuid=True), ForeignKey("company_profiles.id"), nullable=False)
    
    # Escopo da política
    name = Column(String(255), nullable=False)  # "Política Tech Sênior"
    department = Column(String(100), nullable=True)  # null = aplica a todos
    role_pattern = Column(String(200), nullable=True)  # "desenvolvedor|engenheiro|arquiteto"
    seniority_level = Column(String(50), nullable=True)  # "senior", "pleno", etc.
    
    # Faixas Salariais
    salary_min = Column(Float, nullable=True)  # R$ 18.000
    salary_max = Column(Float, nullable=True)  # R$ 25.000
    salary_target = Column(Float, nullable=True)  # R$ 22.000 (ponto médio desejado)
    
    # Política de Bônus
    bonus_enabled = Column(Boolean, default=False)
    bonus_type = Column(String(50), nullable=True)  # "annual", "quarterly", "on-target"
    bonus_min_pct = Column(Float, nullable=True)  # 10% (mínimo)
    bonus_target_pct = Column(Float, nullable=True)  # 20% (target)
    bonus_max_pct = Column(Float, nullable=True)  # 40% (máximo)
    bonus_criteria = Column(Text, nullable=True)  # Descrição dos critérios
    
    # Remuneração Variável (além de bônus)
    variable_compensation = Column(JSON, default={})
    # {
    #   "plr": {"enabled": true, "avg_pct": 15},
    #   "comissao": {"enabled": false},
    #   "stock_options": {"enabled": true, "vesting": "4 anos"}
    # }
    
    # Total Compensation Target
    total_comp_annual_min = Column(Float, nullable=True)
    total_comp_annual_max = Column(Float, nullable=True)
    
    # Metadados
    is_active = Column(Boolean, default=True)
    effective_from = Column(DateTime, default=datetime.utcnow)
    effective_until = Column(DateTime, nullable=True)
    source = Column(String(100), default="manual")  # manual, imported, ai_suggested
    
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    created_by = Column(String(255), nullable=True)

class CompensationAnalysisService:
    """
    Serviço de análise proativa de compensação.
    Combina dados de mercado, políticas da empresa e histórico interno.
    """
    
    def __init__(self):
        self.market_benchmark = MarketBenchmarkService()
        self.company_config = CompanyConfigurationService()
        self.job_insights = JobInsightsService()
    
    async def analyze_compensation(
        self,
        company_id: str,
        job_title: str,
        seniority: str,
        department: Optional[str] = None,
        location: Optional[str] = None,
        proposed_salary_min: Optional[float] = None,
        proposed_salary_max: Optional[float] = None,
        proposed_bonus_pct: Optional[float] = None,
        proposed_benefits: Optional[List[str]] = None
    ) -> CompensationAnalysisResult:
        """
        Analisa proposta de compensação comparando com:
        1. Política da empresa (se cadastrada)
        2. Dados de mercado
        3. Histórico interno de vagas similares
        """
        
        # 1. Buscar política da empresa para este cargo/nível
        policy = await self._get_company_policy(
            company_id, job_title, seniority, department
        )
        
        # 2. Buscar benchmark de mercado
        market_data = await self.market_benchmark.search_salary_benchmark(
            role=job_title,
            seniority=seniority,
            location=location
        )
        
        # 3. Buscar histórico interno
        internal_data = await self.job_insights.get_salary_insights(
            company_id=company_id,
            role_pattern=job_title,
            seniority=seniority
        )
        
        # 4. Buscar benefícios da empresa
        company_config = await self.company_config.get_configuration(company_id)
        company_benefits = company_config.get_benefits_for_seniority(seniority)
        
        # 5. Analisar cada componente
        salary_analysis = self._analyze_salary(
            proposed_min=proposed_salary_min,
            proposed_max=proposed_salary_max,
            policy=policy,
            market=market_data,
            internal=internal_data
        )
        
        bonus_analysis = self._analyze_bonus(
            proposed_pct=proposed_bonus_pct,
            policy=policy
        )
        
        benefits_analysis = self._analyze_benefits(
            proposed=proposed_benefits or [],
            company_standard=company_benefits,
            seniority=seniority
        )
        
        # 6. Calcular Total Compensation
        total_comp = self._calculate_total_comp(
            salary_analysis,
            bonus_analysis,
            benefits_analysis,
            policy
        )
        
        # 7. Gerar resumo e recomendações
        return self._build_result(
            salary_analysis,
            bonus_analysis,
            benefits_analysis,
            total_comp,
            policy
        )

    def _analyze_salary(self, ...) -> SalaryAnalysis:
        """Analisa componente salarial"""
        analysis = SalaryAnalysis(
            proposed_min=proposed_min or 0,
            proposed_max=proposed_max or 0
        )
        
        # Comparar com mercado
        if market and market.get("min"):
            analysis.market_min = market["min"]
            analysis.market_max = market["max"]
            analysis.market_median = market.get("median")
            analysis.data_sources.append(DataSource.MARKET_BENCHMARK)
            
            # Determinar alinhamento
            if proposed_max and proposed_max < market["min"] * 0.9:
                analysis.market_alignment = CompensationAlignmentStatus.BELOW_MARKET
            elif proposed_min and proposed_min > market["max"] * 1.1:
                analysis.market_alignment = CompensationAlignmentStatus.ABOVE_MARKET
            else:
                analysis.market_alignment = CompensationAlignmentStatus.ALIGNED
        
        # Comparar com política da empresa
        if policy:
            analysis.policy_min = policy.salary_min
            analysis.policy_max = policy.salary_max
            analysis.data_sources.append(DataSource.COMPANY_POLICY)
            
            # Verificar alinhamento com política
            if proposed_min and proposed_min < policy.salary_min:
                analysis.policy_alignment = CompensationAlignmentStatus.BELOW_POLICY
            elif proposed_max and proposed_max > policy.salary_max:
                analysis.policy_alignment = CompensationAlignmentStatus.ABOVE_POLICY
            else:
                analysis.policy_alignment = CompensationAlignmentStatus.ALIGNED
        
        # Gerar sugestão se necessário
        if analysis.market_alignment == CompensationAlignmentStatus.BELOW_MARKET:
            analysis.suggested_min = market["min"]
            analysis.suggested_max = market.get("median") or market["max"]
            analysis.suggestion_reason = (
                f"Sugerimos ajustar para a faixa de mercado ({format_brl(market['min'])} - {format_brl(market['max'])})"
            )
        
        return analysis
    
    def _calculate_monetizable_benefits(
        self,
        benefits: List[Dict[str, Any]]
    ) -> Tuple[float, List[Dict[str, Any]]]:
        """Calcula valor anual de benefícios monetizáveis"""
        breakdown = []
        total = 0.0
        
        for benefit in benefits:
            if benefit.get("value") and benefit.get("value_type") == "monetary":
                monthly = benefit["value"]
                annual = monthly * 12
                breakdown.append({
                    "name": benefit["name"],
                    "monthly": monthly,
                    "annual": annual
                })
                total += annual
        
        return total, breakdown
