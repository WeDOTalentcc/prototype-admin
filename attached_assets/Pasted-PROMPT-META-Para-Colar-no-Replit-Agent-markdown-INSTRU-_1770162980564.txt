````markdown
1. job_description → Criar JD
2. competencies → Definir competências + remuneração

[Sua ação: chamada de tool OU pergunta ao usuário]
````

````javascript
import Anthropic from '@anthropic-ai/sdk';

const client = new Anthropic({

// Função que CHAMA um agente especializado
async function callAgent(agentName, input, agentSystemPrompt) {
  const response = await client.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 4096,
    system: agentSystemPrompt, // Prompt específico do agente
    messages: [
      {
        role: "user",
        content: JSON.stringify(input)
      }
    ]
  });

  // Agentes SEMPRE retornam JSON
  const agentOutput = JSON.parse(response.content[0].text);
  return agentOutput;
}

// Função principal do Orchestrator
export async function processWithOrchestrator(userMessage, conversationHistory, currentState) {
  
  // 1. Carregar o system prompt do orchestrator
  const orchestratorPrompt = `${ORCHESTRATOR_SYSTEM_PROMPT}

ESTADO ATUAL:
currentStep: "${currentState.currentStep}"
completedSteps: ${JSON.stringify(currentState.completedSteps)}
jobData: ${JSON.stringify(currentState.jobData)}
`;

  // 2. Fazer chamada para o Orchestrator com tools
  let response = await client.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 4096,
    system: orchestratorPrompt,
    tools: ORCHESTRATOR_TOOLS,
    messages: [
      ...conversationHistory,
      {
        role: "user",
        content: userMessage
      }
    ]
  });

  // 3. Loop de tool use (agentic behavior)
  const toolResults = [];
  
  while (response.stop_reason === "tool_use") {
    const toolUse = response.content.find(block => block.type === "tool_use");
    
    let toolResult;
    
    // 4. Executar a tool que o Orchestrator pediu
    switch (toolUse.name) {
      case "call_jd_agent":
        toolResult = await callAgent(
          "jd_agent",
          toolUse.input,
          JD_AGENT_SYSTEM_PROMPT // Importar de prompts/jd_agent.md
        );
        break;
        
      case "call_competencies_agent":
        toolResult = await callAgent(
          "competencies_agent",
          toolUse.input,
          COMPETENCIES_AGENT_SYSTEM_PROMPT
        );
        break;
        
      case "call_questions_agent":
        toolResult = await callAgent(
          "questions_agent",
          toolUse.input,
          QUESTIONS_AGENT_SYSTEM_PROMPT
        );
        break;
        
      case "mark_step_complete":
        // Atualizar estado
        currentState.completedSteps.push(toolUse.input.stepName);
        currentState.currentStep = getNextStep(toolUse.input.stepName);
        toolResult = { success: true, newStep: currentState.currentStep };
        break;
    }
    
    // 5. Retornar resultado da tool para o Orchestrator
    response = await client.messages.create({
      model: "claude-sonnet-4-20250514",
      max_tokens: 4096,
      system: orchestratorPrompt,
      tools: ORCHESTRATOR_TOOLS,
      messages: [
        ...conversationHistory,
        { role: "user", content: userMessage },
        { role: "assistant", content: response.content },
        {
          role: "user",
          content: [
            {
              type: "tool_result",
              tool_use_id: toolUse.id,
              content: JSON.stringify(toolResult)
            }
          ]
        }
      ]
    });
  }
  
  // 6. Retornar resposta final do Orchestrator
  return {
    response: response.content[0].text,
    updatedState: currentState
  };
}

function getNextStep(currentStep) {
  const flow = ["job_description", "competencies", "benchmark", "questions", "review", "complete"];
  const currentIndex = flow.indexOf(currentStep);
  return flow[currentIndex + 1] || "complete";
}
````

#### D. API Endpoint

Arquivo: `api/chat.js`
````javascript
import { processWithOrchestrator } from '../src/llm-integration.js';

// Em memória (depois usar Redis/DB)
const sessions = new Map();

export default async function handler(req, res) {
  const { message, sessionId } = req.body;
  
  // Carregar ou criar sessão
  if (!sessions.has(sessionId)) {
    sessions.set(sessionId, {
      conversationHistory: [],
      state: {
        currentStep: "job_description",
        completedSteps: [],
        jobData: {}
      }
    });
  }
  
  const session = sessions.get(sessionId);
  
  // Processar com Orchestrator
  const result = await processWithOrchestrator(
    message,
    session.conversationHistory,
    session.state
  );
  
  // Atualizar histórico
  session.conversationHistory.push(
    { role: "user", content: message },
    { role: "assistant", content: result.response }
  );
  
  session.state = result.updatedState;
  
  res.json({
    response: result.response,
    currentStep: result.updatedState.currentStep,
    progress: `${result.updatedState.completedSteps.length}/5 etapas`
  });
}
````
