```javascript
// Design System LIA v4.1 - Tokens para React + Tailwind

export const colors = {
  // Backgrounds
  bg: {
    primary: 'bg-white',
    secondary: 'bg-gray-50',
    tertiary: 'bg-gray-100',
  },
  
  // Texts
  text: {
    primary: 'text-gray-900',
    body: 'text-gray-800',
    secondary: 'text-gray-600',
    muted: 'text-gray-500',
    disabled: 'text-gray-400',
  },
  
  // Borders
  border: {
    subtle: 'border-gray-200',
    default: 'border-gray-300',
    medium: 'border-gray-400',
  },
  
  // WeDo Accent Colors (10% - apenas contextos)
  wedo: {
    cyan: '#60BED1',      // Brain icon LIA
    green: '#5DA47A',     // Candidatos
    orange: '#D19960',    // Tempo
    purple: '#9860D1',    // Insights
    magenta: '#D160AB',   // Crítico
  },
}

export const spacing = {
  0: 'p-0',
  1: 'p-2',      // 8px
  2: 'p-4',      // 16px
  3: 'p-6',      // 24px
  4: 'p-8',      // 32px
  5: 'p-10',     // 40px
  6: 'p-12',     // 48px
}

// Helper para gerar classes de botões
export function getButtonClasses(variant = 'primary', size = 'default') {
  const baseClasses = 'font-semibold rounded transition-all duration-150 focus:outline-none focus:ring-2 focus:ring-gray-900/20'
  
  const variants = {
    primary: 'bg-gray-900 text-white hover:bg-gray-800 active:bg-gray-700',
    secondary: 'bg-transparent text-gray-900 border border-gray-300 hover:bg-gray-50 hover:border-gray-400',
    ghost: 'bg-transparent text-gray-700 hover:bg-gray-100 hover:text-gray-900',
    destructive: 'bg-red-600 text-white hover:bg-red-700 active:bg-red-800',
  }
  
  const sizes = {
    small: 'px-3 py-1.5 text-xs',
    default: 'px-4 py-2 text-sm',
    large: 'px-5 py-2.5 text-sm',
  }
  
  return `${baseClasses} ${variants[variant]} ${sizes[size]}`
}

// Helper para badges
export function getBadgeClasses(type = 'neutral') {
  const base = 'inline-flex items-center px-2 py-1 rounded text-xs font-medium border'
  
  const types = {
    success: 'bg-green-50 text-green-700 border-green-200',
    warning: 'bg-amber-50 text-amber-700 border-amber-200',
    error: 'bg-red-50 text-red-700 border-red-200',
    info: 'bg-blue-50 text-blue-700 border-blue-200',
    neutral: 'bg-gray-100 text-gray-700 border-gray-200',
  }
  
  return `${base} ${types[type]}`
}

// Helper para inputs
export function getInputClasses(hasError = false) {
  const base = 'w-full px-3 py-2 text-sm text-gray-900 bg-white border rounded transition-all duration-150 focus:outline-none'
  const normal = 'border-gray-300 focus:border-gray-900 focus:ring-2 focus:ring-gray-900/20'
  const error = 'border-red-500 focus:ring-2 focus:ring-red-500/20'
  
  return `${base} ${hasError ? error : normal}`
}
```

module.exports = {
  content: [
    "./src/**/*.{js,jsx,ts,tsx}",
  ],
  theme: {
    extend: {
      colors: {
        // Manter cores Tailwind padrão (gray-50 até gray-900)
        // Adicionar cores WeDo como custom
        'wedo-cyan': '#60BED1',
        'wedo-green': '#5DA47A',
        'wedo-orange': '#D19960',
        'wedo-purple': '#9860D1',
        'wedo-magenta': '#D160AB',
      },
      fontFamily: {
        'brand': ['Open Sans', 'sans-serif'],  // Headers, buttons, labels
        'data': ['Inter', 'sans-serif'],        // Body, forms, tables
      },
      spacing: {
        // Sistema 8px (Tailwind já usa por padrão, mas garantir)
        // 2 = 8px, 4 = 16px, 6 = 24px, 8 = 32px
      },
      boxShadow: {
        'focus': '0 0 0 3px rgba(17, 24, 39, 0.2)', // Focus ring padrão
