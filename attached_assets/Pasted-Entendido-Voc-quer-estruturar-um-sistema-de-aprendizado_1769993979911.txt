Como o Wizard Usa Cada Camada
Prioridade de Fontes (Reasoning do Orquestrador)
class WizardIntelligenceLayer:
    """
    Camada de inteligência que decide qual fonte usar para cada campo.
    """
    
    PRIORITY_ORDER = [
        "workforce_planning",  # 1. Mais autoritativo - dados planejados
        "client_history",      # 2. Padrões do cliente
        "curated_templates",   # 3. Templates WSI (662)
        "llm_inference",       # 4. Fallback - LLM gera
    ]
    
    def get_suggestion(self, field: str, context: JobContext) -> Suggestion:
        for source in self.PRIORITY_ORDER:
            suggestion = self.query_source(source, field, context)
            if suggestion and suggestion.confidence > 0.7:
                return Suggestion(
                    value=suggestion.value,
                    source=source,
                    confidence=suggestion.confidence,
                    explanation=f"Baseado em {source}"
                )
        
        # Fallback: LLM gera com base no contexto
        return self.llm_generate(field, context)

Resumo da Lógica do Wizard
class WizardDataPriority:
    """
    Ordem de consulta para cada campo no wizard.
    """
    
    PRIORITY = [
        "settings",           # 1. Menu Configurações
        "lia_history",        # 2. Histórico da plataforma LIA
        "ats_basic_import",   # 3. JDs importados (básico)
        "workforce_planning", # 4. HRIS/Workforce Planning
        "ats_full_import",    # 5. ATS completo + ETLs
        "curated_templates",  # 6. Templates curados (fallback)
    ]
    
    def get_field_value(self, field: str, context: JobContext):
        for source in self.PRIORITY:
            value = self.query(source, field, context)
            if value and value.confidence > 0.7:
                return value
        return self.llm_fallback(field, context)
