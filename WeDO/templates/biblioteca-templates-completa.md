## Estrutura do Banco de Dados

```sql
CREATE TABLE job_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID NULL, -- NULL = template do sistema
    
    -- Categorização
    category VARCHAR(50) NOT NULL, -- 'tecnologia', 'financas', etc
    subcategory VARCHAR(50) NOT NULL, -- 'desenvolvimento', 'dados', etc
    
    -- Informações básicas
    title VARCHAR(200) NOT NULL,
    title_normalized VARCHAR(200) NOT NULL, -- para busca
    seniority VARCHAR(20) NOT NULL, -- 'junior', 'pleno', 'senior', 'gestao', 'c-level'
    
    -- Conteúdo padrão
    default_description TEXT,
    default_requirements TEXT,
    default_skills JSONB DEFAULT '[]', -- skills técnicas sugeridas
    default_behavioral JSONB DEFAULT '[]', -- competências comportamentais
    
    -- Configurações de faixa salarial
    salary_range_min INTEGER,
    salary_range_max INTEGER,
    salary_currency VARCHAR(3) DEFAULT 'BRL',
    
    -- Configurações de trabalho
    work_model VARCHAR(20) DEFAULT 'hybrid', -- remote/hybrid/onsite
    employment_type VARCHAR(20) DEFAULT 'clt', -- clt/pj/temporary
    
    -- Metadados
    is_system BOOLEAN DEFAULT FALSE, -- template do sistema vs empresa
    is_active BOOLEAN DEFAULT TRUE,
    usage_count INTEGER DEFAULT 0,
    popularity_score FLOAT DEFAULT 0,
    
    -- Auditoria
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    created_by UUID
);

-- Índices
CREATE INDEX idx_templates_category ON job_templates(category);
CREATE INDEX idx_templates_subcategory ON job_templates(subcategory);
CREATE INDEX idx_templates_title_normalized ON job_templates(title_normalized);
CREATE INDEX idx_templates_company ON job_templates(company_id);
CREATE INDEX idx_templates_popularity ON job_templates(popularity_score DESC);
```
